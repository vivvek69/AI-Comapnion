#
#   sudo apt update && sudo apt upgrade -y
#   sudo apt install -y \
#       python3-pip python3-venv \
#       espeak-ng libespeak-ng1 \
#       libatlas-base-dev libhdf5-dev \
#       libgtk-3-dev libopencv-dev \

#       libjpeg-dev libpng-dev libtiff-dev \
#       alsa-utils libasound2-dev \
#       bluez bluez-alsa-utils
#
# Bluetooth speaker setup (BlueZ + bluealsa):
#   1. Pair and trust your speaker:
#        bluetoothctl
#        > scan on
#        > pair AA:BB:CC:DD:EE:FF
#        > trust AA:BB:CC:DD:EE:FF
#        > connect AA:BB:CC:DD:EE:FF
#   2. Start the bluealsa ALSA daemon:
#        sudo systemctl enable --now bluealsa
#   3. Verify the ALSA device is present:
#        aplay -L | grep bluealsa
#   4. (Optional) Set a specific MAC in your .env:
#        BT_SPEAKER_ALSA_DEVICE=bluealsa:DEV=AA:BB:CC:DD:EE:FF,PROFILE=a2dp
#
# USB mic / webcam:
#   - Plug in and run `arecord -l` to confirm the USB mic appears.
#   - Run `v4l2-ctl --list-devices` to confirm the webcam is /dev/video0.
#
# Then:
#   python3 -m venv venv
#   source venv/bin/activate
#   pip install -r requirements-pi.txt
